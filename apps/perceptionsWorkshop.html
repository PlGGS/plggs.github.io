Perceptions Workshop is an application I wrote for a multi-school Operation Snowball that took place towards the end of my senior year of high school. In order to make sure there weren't any technical issues with Google Drive, I created a basic offline slide player. This allowed multiple groups to teach about different perceptions without the need for an internet connection of any kind. It also allowed me to help out with the event after already having been a Director.

using AxWMPLib;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace PerceptionsWorkshop
{
    public class SlideManager
    {
        public static void KeyUp(Form currentForm, KeyEventArgs e)
        {
            if (e.KeyCode == Keys.Space || e.KeyCode == Keys.Right)
            {
                LoadNextForm(currentForm);
            }
            else if (e.KeyCode == Keys.Left)
            {
                LoadLastForm(currentForm);
            }
            else if (e.KeyCode == Keys.Escape)
            {
                Application.Exit();
            }
        }

        public static void KeyUp(Form currentForm, _WMPOCXEvents_KeyUpEvent e)
        {
            if (e.nKeyCode == (short)Keys.Space || e.nKeyCode == (short)Keys.Right)
            {
                LoadNextForm(currentForm);
            }
            else if (e.nKeyCode == (short)Keys.Left)
            {
                LoadLastForm(currentForm);
            }
            else if (e.nKeyCode == (short)Keys.Escape)
            {
                Application.Exit();
            }
        }

        public static void LoadNextForm(Form currentForm)
        {
            Form frm = null;

            if (currentForm is frmRead)
            {
                frm = new frmListen0();
            }
            else if (currentForm is frmListen0)
            {
                frm = new frmListen1();
            }
            else if (currentForm is frmListen1)
            {
                frm = new frmWatch();
            }
            else if (currentForm is frmWatch)
            {
                frm = new frmListen2();
            }
            else if (currentForm is frmListen2)
            {
                frm = new frmListen3();
            }
            else if (currentForm is frmListen3)
            {
                frm = new frmListen4();
            }
            else if (currentForm is frmListen4)
            {
                frm = new frmListen4();
            }

            if (frm != null)
            {
                frm.Show();
                currentForm.Close();
            }
        }

        public static void LoadLastForm(Form currentForm)
        {
            Form frm = null;

            if (currentForm is frmListen0)
            {
                frm = new frmRead();
            }
            else if (currentForm is frmListen1)
            {
                frm = new frmListen0();
            }
            else if (currentForm is frmWatch)
            {
                frm = new frmListen1();
            }
            else if (currentForm is frmListen2)
            {
                frm = new frmWatch();
            }
            else if (currentForm is frmListen3)
            {
                frm = new frmListen2();
            }
            else if (currentForm is frmListen4)
            {
                frm = new frmListen3();
            }

            if (frm != null)
            {
                frm.Show();
                currentForm.Close();
            }
        }
    }
}
