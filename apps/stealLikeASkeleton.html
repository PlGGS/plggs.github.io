Steal Like a Skeleton is an application I wrote for Spanish class during my junior year of high school. It allowed each and every student in the class to create their own personal skeleton skull for the Spanish holiday, The Day of the Dead. The application's background music solved my teacher's problem of not having Spotify premium at the time. The class loved the idea, and I was finally able to apply an object oriented application perspective to an application of my own!

using System;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;

namespace Steal_Like_a_Skeleton
{
    public partial class frmSkull : Form
    {
        string skullName;
        string skullEyes;
        string skullNose;
        string skullMouth;

        public frmSkull(string name, string eyes, string nose, string mouth)
        {
            skullName = name;
            skullEyes = eyes;
            skullNose = nose;
            skullMouth = mouth;

            Console.WriteLine(name + ", " + eyes + ", " + nose + ", " + mouth);

            InitializeComponent();
        }

        private void frmSkull_Load(object sender, EventArgs e)
        {
            lblSkullName.Text = skullName;
            picEyes.Image = Dictionaries.EyeShapes().FirstOrDefault(x => x.Value == skullEyes).Key;
            picNose.Image = Dictionaries.NoseShapes().FirstOrDefault(x => x.Value == skullNose).Key;
            picMouth.Image = Dictionaries.MouthShapes().FirstOrDefault(x => x.Value == skullMouth).Key;

            this.Location = new Point(Screen.PrimaryScreen.Bounds.Width / 2 - this.Bounds.Width / 2, Screen.PrimaryScreen.Bounds.Height / 2 - this.Bounds.Height / 2);
            lblSkullName.Left = this.Bounds.Width / 2 - lblSkullName.Width / 2;
            this.Text = skullName;
        }
        
        private void frmSkull_KeyUp(object sender, KeyEventArgs e)
        {
            if (e.KeyCode == Keys.Space)
            {
                frmMain main = new frmMain(true);
                main.Show();
            }

            if (e.KeyCode == Keys.Escape)
            {
                Application.Exit();
            }
        }
    }
}
