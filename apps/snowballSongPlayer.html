Snowball Song Player is an application I wrote in a period of two days in preparation for Operation Snowball during my senior year of high school. This application allowed myself and the other directors to play each and every song we needed for skits and the lip sync battle with ease! It also allowed me to easily give the small groups their lip sync battle songs by randomly selecting from the available songs and assigning them to each group. This made two of the most technologically difficult parts of Snowball so much easier and reliable than they had been in years past.

        }

        }

            wmpMain.URL = "Audio\\LGBTQ+ - Birds Set Free.wav";

        {
            wmpMain.URL = "Audio\\Grief - Only One.wav";
        }

        {
            wmpMain.URL = "Audio\\Mental Health - Unsteady.wav";
        }

        private void btnBell_Click(object sender, EventArgs e)
        {
            time = wmpMain.Ctlcontrols.currentPosition;
            fileName = wmpMain.URL;
            bellPlaying = true;

            wmpMain.URL = "Audio\\Bell.wav";
        }

        private void btnLipSync_Click(object sender, EventArgs e)
        {
            int item = -1;

            //loop through lbx to find instance of group name
            for (int i = 0; i < lbxGroupSongs.Items.Count; i++)
            {
                if (lbxGroupSongs.Items[i].ToString().Split(':')[0] == cbxSmallGroups.Text)
                {
                    item = i;
                }
            }

            //set url to lbxGroupSongs.Items[
            BeginInvoke(new Action(() => {
                wmpMain.URL = usedSongNamesAndFiles[lbxGroupSongs.Items[item].ToString().Split(':')[1].Substring(1)];
            }));
        }

        private void wmpMain_PlayStateChange(object sender, AxWMPLib._WMPOCXEvents_PlayStateChangeEvent e)
        {
            if (wmpMain.playState == WMPLib.WMPPlayState.wmppsMediaEnded)
            {
                Console.WriteLine(wmpMain.URL + " | " + wmpMain.playState);
                if (bellPlaying == true)
                {
                    BeginInvoke(new Action(() => {
                        wmpMain.URL = fileName;
                        wmpMain.Ctlcontrols.currentPosition = time;
                    }));

                    bellPlaying = false;
                }
            }
        }

        private void wmpMain_EndOfStream(object sender, AxWMPLib._WMPOCXEvents_EndOfStreamEvent e)
        {
            
        }

        private void btnGetSong_Click(object sender, EventArgs e)
        {
            bool used = false;
            StreamReader sr = new StreamReader(groupSongsFilePath);
            string line = "";

            for (int i = 0; i < sr.BaseStream.Length; i++)
            {
                line = sr.ReadLine();
                if (line != null)
                {
                    if (cbxSmallGroups.SelectedItem.ToString() == line.Split(':')[0])
                    {
                        used = true;
                    }
                }
            }

            sr.Close();

            if (used == false)
            {
                StreamWriter sw = new StreamWriter(groupSongsFilePath, true);
                wmpMain.URL = "Audio\\SelectSong.wav";
                Random rnd = new Random();
                int songNum = rnd.Next(0, songNamesAndFiles.Values.Count - 1);

                sw.WriteLine(cbxSmallGroups.Text + ": " + songNamesAndFiles.Keys.ToArray()[songNum] + "\n");

                if (groupSongs == null)
                    groupSongs = new List<string>();

                MessageBox.Show(songNamesAndFiles.Keys.ToArray()[songNum], "Your song!");
                
                sw.Close();
                Console.WriteLine(songNamesAndFiles.Values.Count);
                usedSongNamesAndFiles.Add(songNamesAndFiles.Keys.ToArray()[songNum], songNamesAndFiles[songNamesAndFiles.Keys.ToArray()[songNum]]); //add song to usedSongs
                songNamesAndFiles.Remove(songNamesAndFiles.Keys.ToArray()[songNum]); //remove song from songNames
                Console.WriteLine(songNamesAndFiles.Values.Count);

                SetGroupSongs();
            }
            else
            {
                MessageBox.Show("Please choose a group that does not already have a song!", "Whoops!");
            }
        }

        private void SetGroupSongs()
        {
            StreamReader sr = new StreamReader(groupSongsFilePath);
            if (groupSongs == null)
                groupSongs = new List<string>();

            lbxGroupSongs.Items.Clear();

            for (int i = 0; i < sr.BaseStream.Length; i++)
            {
                string line = sr.ReadLine();
                if (line != null && line != "")
                {
                    groupSongs.Add(line);
                    if (!usedSongNamesAndFiles.Keys.Contains(line.Split(':')[1].Substring(1)) &&
                        !usedSongNamesAndFiles.Values.Contains("Audio\\" + line.Split(':')[1].Substring(1) + ".wav"))
                    {
                        usedSongNamesAndFiles.Add(line.Split(':')[1].Substring(1), "Audio\\" + line.Split(':')[1].Substring(1) + ".wav"); //add song to usedSongs

                    }
                    if (songNamesAndFiles.Keys.Contains(songNamesAndFiles.Keys.Contains(line.Split(':')[1].Substring(1)).ToString()))
                    {
                        songNamesAndFiles.Remove(songNamesAndFiles.Keys.Contains(line.Split(':')[1].Substring(1)).ToString()); //remove song from songNames
                    }
                    lbxGroupSongs.Invoke(new MethodInvoker(delegate { lbxGroupSongs.Items.Add(line); }));
                }
            }

            sr.Close();
        }
    }
