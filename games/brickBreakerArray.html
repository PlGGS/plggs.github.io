Brick Breaker Array is a simple Brick Breaker Style game that used 'randomly' generated Minecraft blocks as bricks. This application combined many programming skills that were quite new to me at the time such as loops, mouse tracking, and file access. Each layer of blocks has a 'randomly' generated amount of ores that count for more points. Each block in the array has different states that correspond to their winforms backcolor properties, because winforms isn't able to compare images.

    Sub paddleCollision(ByRef ball As PictureBox, ByRef paddle As PictureBox)

        If ball.Bounds.IntersectsWith(paddle.Bounds) Then
            ballSpeedY = -ballSpeedY
        End If

    End Sub

    Sub placeBlocks(ByRef ray(,) As PictureBox)

        Randomize()
        Dim Chance As Integer

        For x As Integer = 0 To ray.GetUpperBound(0)

            xBlock += 32

            For y As Integer = 0 To ray.GetUpperBound(1)
                Chance = Int(Rnd() * 20 + 1)
                block = New PictureBox
                block.Name = x & y
                block.Height = 32
                block.Width = 32
                block.Top = xBlock
                block.Left = yBlock
                block.BackColor = Color.Gray

                'below is spawning ores

                If x <= 3 And Chance <= 1 Then
                    block.BackgroundImage = My.Resources.diamond_ore
                    block.BackColor = Color.Blue
                Else
                    block.BackgroundImage = My.Resources.stone
                    block.BackColor = Color.Gray
                End If

                If x <= 6 And x >= 3 And Chance <= 3 Then
                    block.BackgroundImage = My.Resources.redstone_ore
                    block.BackColor = Color.Red
                ElseIf x <= 6 And x > 3 Then
                    block.BackgroundImage = My.Resources.stone
                    block.BackColor = Color.Gray
                End If

                If x <= 9 And x >= 6 And Chance <= 3 Then
                    block.BackgroundImage = My.Resources.gold_ore
                    block.BackColor = Color.Yellow
                ElseIf x <= 9 And x >= 6 Then
                    block.BackgroundImage = My.Resources.stone
                    block.BackColor = Color.Gray
                End If

                If x <= 11 And x >= 9 And Chance <= 3 Then
                    block.BackgroundImage = My.Resources.iron_ore
                    block.BackColor = Color.PaleGreen
                ElseIf x <= 11 And x >= 9 Then
                    block.BackgroundImage = My.Resources.stone
                    block.BackColor = Color.Gray
                End If

                If x <= 12 And x >= 11 And Chance <= 4 Then
                    block.BackgroundImage = My.Resources.coal_ore
                    block.BackColor = Color.Black
                ElseIf x <= 12 And x >= 11 Then
                    block.BackgroundImage = My.Resources.stone
                    block.BackColor = Color.Gray
                End If

                block.BackgroundImageLayout = ImageLayout.Zoom

                ray(x, y) = block
                Controls.Add(block)
                yBlock += 32

            Next

            yBlock = 10

        Next

    End Sub

    Sub blockCollision(ByRef ray(,) As PictureBox, ByRef ball As PictureBox)

        For x As Integer = 0 To ray.GetUpperBound(0)

            For y As Integer = 0 To ray.GetUpperBound(1)

                'below is hitting stone and cobble

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.Gray And ray(x, y).Visible = True Then

                    Score += 10

                    ballSpeedY = -ballSpeedY

                    ray(x, y).BackgroundImage = My.Resources.cobblestone
                    ray(x, y).BackgroundImageLayout = ImageLayout.Zoom
                    ray(x, y).BackColor = Color.DarkGray
                    ball.Top += 32

                ElseIf ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.DarkGray And ray(x, y).Visible = True Then

                    Score += 10

                    ray(x, y).Visible = False

                End If

                'below is hitting coal

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.Black And ray(x, y).Visible = True Then

                    Score += 20

                    ballSpeedY = -ballSpeedY

                    ball.Top += 32

                    ray(x, y).Visible = False

                End If

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.PaleGreen And ray(x, y).Visible = True Then

                    Score += 30

                    ballSpeedY = -ballSpeedY

                    ball.Top += 32

                    ray(x, y).Visible = False

                End If

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.Yellow And ray(x, y).Visible = True Then

                    Score += 40

                    ballSpeedY = -ballSpeedY

                    ball.Top += 32

                    ray(x, y).Visible = False

                End If

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.Red And ray(x, y).Visible = True Then

                    Score += 50

                    ballSpeedY = -ballSpeedY

                    ball.Top += 32

                    ray(x, y).Visible = False

                End If

                If ball.Bounds.IntersectsWith(ray(x, y).Bounds) And ray(x, y).BackColor = Color.Blue And ray(x, y).Visible = True Then

                    Score += 100

                    ballSpeedY = -ballSpeedY

                    ball.Top += 32

                    ray(x, y).Visible = False

                End If

            Next

        Next

    End Sub

    Private Sub picBall_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles picBall.Click

        If Lives = 3 Then

            timScoreAndLives.Enabled = True
            timTime.Enabled = True
            timBallCollision.Enabled = True
            timBallMove.Enabled = True

            placeBlocks(blocks)

        ElseIf Lives = 2 Or Lives = 1 Then

            timScoreAndLives.Enabled = True
            timTime.Enabled = True
            timBallCollision.Enabled = True
            timBallMove.Enabled = True

        End If

    End Sub

    Private Sub tmeBallMove_Tick(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles timBallMove.Tick

        BallMove(picBall, ballSpeedX, ballSpeedY)

    End Sub

    Private Sub tmeBallCollision_Tick(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles timBallCollision.Tick

        checkWallCollision(picBall)
        paddleCollision(picBall, picPaddle)
        blockCollision(blocks, picBall)

    End Sub

    Private Sub timTime_Tick(sender As Object, e As EventArgs) Handles timTime.Tick

        lblTime.Text = "Time: " & Time

        Time += 1

    End Sub

    Private Sub timScoreAndLives_Tick(sender As Object, e As EventArgs) Handles timScoreAndLives.Tick

        lblScore.Text = "Score: " & Score

        lblLives.Text = "Lives: " & Lives

    End Sub
